\question[8]
Solve the equation $\frac{2x}{x-5} = 3 + \frac{1-x}{x-3}$.
\vspace*{5cm}

\question[12]
Consider the function $g(x) = \frac{3-2x}{x-7}$.
\begin{parts}
    \part Find the domain of $g$.
    \vspace*{2cm}
    
    \part Find the range of $g$.
    \vspace*{3cm}
    
    \part Find all horizontal and vertical asymptotes of the graph of $y = g(x)$.
    \vspace*{3cm}
    
    \part Graph $y = g(x)$.
    \vspace*{4cm}
\end{parts}

\newpage

\question[10]
Find constants $A$ and $B$ such that $\frac{x+7}{x^2-2x-35} = \frac{A}{x-7} + \frac{B}{x+5}$ for all $x$.
\vspace*{5cm}

\question[12]
Find the constants $A$, $B$, and $C$ such that 
$$\frac{3x-2}{x^3+2x^2+x} = \frac{A}{x} + \frac{B}{x+1} + \frac{C}{(x+1)^2}.$$
\vspace*{6cm}

\newpage

\question[8]
Find all values of $t$ that satisfy $\frac{t-7}{t^2-2t-3} = \frac{t+2}{t^2+2t-15}$.
\vspace*{5cm}

\question[10]
Graph each function. Show any asymptotes as dashed lines.
\begin{parts}
    \part $h(x) = \frac{x+1}{x-1}$
    \vspace*{4cm}
    
    \part $h(x) = \frac{10x^2}{x^2+1}$
    \vspace*{4cm}
\end{parts}

\newpage

\question[15]
For the rational function $f(x) = \frac{x^2-4}{x^2-3x-4}$:
\begin{parts}
    \part Find the domain of $f$.
    \vspace*{2cm}
    
    \part Find all vertical asymptotes.
    \vspace*{2cm}
    
    \part Find all horizontal asymptotes.
    \vspace*{2cm}
    
    \part Find all holes (removable discontinuities), if any.
    \vspace*{3cm}
    
    \part Find the $x$-intercepts and $y$-intercept.
    \vspace*{3cm}
\end{parts}

\question[10]
Find the oblique asymptote for $f(x) = \frac{2x^2+3x-1}{x+2}$ using polynomial long division.
\vspace*{5cm}

\newpage

\question[8]
Simplify each rational expression:
\begin{parts}
    \part $\frac{x^2-9}{x^2+6x+9}$
    \vspace*{2cm}
    
    \part $\frac{2x^3-8x}{x^2-4}$
    \vspace*{2cm}
    
    \part $\frac{x^3+8}{x^2-4}$
    \vspace*{2cm}
    
    \part $\frac{6x^2-24}{3x^2-6x-24}$
    \vspace*{2cm}
\end{parts}

\question[7]
Perform the indicated operations and simplify:
$$\frac{x^2-1}{x+2} \cdot \frac{x+2}{x-1} \div \frac{x+1}{x-3}$$
\vspace*{4cm}
