\question[8]
Compute $\frac{x}{y}$ if $x + \frac{1}{y} = 4$ and $y + \frac{1}{x} = \frac{1}{4}$.
\vspace*{5cm}

\question[10]
Two lines with nonzero slope and the same $y$-intercept have the property that the sum of their slopes is 0. What is the sum of the $x$-coordinates of their $x$-intercepts?

\textit{Hint:} Turn the words into math. Let the common $y$-intercept be $(0, b)$ and the two $x$-intercepts be $(c, 0)$ and $(d, 0)$.
\vspace*{6cm}

\newpage

\question[12]
A line passing through the points $(2a + 1, 3a^2)$ and $(3a + 1, 5a^2)$ has slope $a + 3$, where $a$ is nonzero. Find the value of $a$.
\vspace*{6cm}



\question[12]
Find $A$ if the graph of the equation $Ax + 3y = 5$ is parallel to the graph of $5x - 2y = 4$.
\vspace*{4cm}

\question[12]
Find $B$ if the graph of the equation $3x - By + 2$ is perpendicular to the graph of $3y = -2x + 4$.
\vspace*{4cm}

\newpage

\question[12]
Find $A$ and $B$ if the graph of the equation $Ax + 3y = B$ produces the same line as the graph of the equation $2x + 6y = 7$.
\vspace*{4cm}

\question[10]
Consider the points $(1, -2)$ and $(-5, 6)$.
\begin{parts}
    \part Find the distance between the two points.
    \vspace*{3cm}
    
    \part What are the coordinates of the midpoint of the segment connecting these points?
    \vspace*{3cm}
    
    \part Find the slope of the line through both points.
    \vspace*{3cm}
    
    \part Find the standard form of the equation whose graph is the line through both points.
    \vspace*{3cm}
\end{parts}

\newpage

\question[10]
If $f(x) = 3x - 2$ and $g(x) = x^2 + 1$, find:
\begin{parts}
    \part $f(g(2))$
    \vspace*{3cm}
    
    \part $g(f(-1))$
    \vspace*{3cm}
\end{parts}

\question[8]
Graph the relation $\{(-3, 1), (-1, 2), (0, -2), (2, 0), (3, 3)\}$.

On your coordinate plane, plot each point. Then answer: Is this relation a function?
\vspace*{6cm}

\question[12]
Give the domain of each relation.
\begin{parts}
    \part $\{(x, y) : y = \frac{2x+1}{x-4}\}$
    \vspace*{2cm}
    
    \part $\{(x, y) : y \leq \frac{3}{x^2-1}\}$
    \vspace*{2cm}
    
    \part $\{(x, y) : y = \frac{x}{|x|+5}\}$
    \vspace*{2cm}
    
    \part $\{(x, y) : |2x - 3| + |y - 1| = 2\}$
    \vspace*{2cm}
\end{parts}

\question[8]
True or False?
\begin{parts}
    \part All relations are functions.
    \vspace*{1cm}
    
    \part All functions are relations.
    \vspace*{1cm}
\end{parts}

\question[10]
Find the range of each function with domain $D$.
\begin{parts}
    \part $h : x \mapsto -2x + 5$, $D = \{1, 3, 5\}$
    \vspace*{2cm}
    
    \part $j : x \mapsto x^2 - 1$, $D = \{-2, 0, 2\}$
    \vspace*{2cm}
    
    \part $k : x \mapsto |x + 2|$, $D = \{-4, -2, 1\}$
    \vspace*{2cm}
    
    \part $m : x \mapsto 4x - 3$, $D = \{0, 2, 4\}$
    \vspace*{2cm}
\end{parts}
